import logging

logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    filename='myapp.log',
    filemode='w'
)

logger = logging.getLogger(__name__)

def divide(x, y):
    try:
        result = x / y
    except ZeroDivisionError:
        logger.error("Tried to divide by zero")
        return None
    return result

result = divide(10, 2)
if result is not None:
    logger.info(f"Result of division: {result}")
